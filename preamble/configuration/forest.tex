%Default styling for semantic tableaus
\newcommand{\semantictableau}[1]{%
    \begin{forest}
        for tree={fit=rectangle, align=center, base=top, s sep=3em, l sep=5ex}
        #1
    \end{forest}
}

%Unable to make use of optional parameters for ID
% as using brackets inside of package "forest" makes it cry
\newcommand{\formula}[3]{%
    $#1$%
    \hspace{0.26em}\color{grey-base}{:}\hspace{0.6em}%
    {\ifthenelse{\equal{#2}{\T}}{\color{green-dark}}{\color{red-dark}}#2}%
    \ifblank{#3}{}{{\hspace{0.8em}\scriptsize \color{blue-dark}{#3}}}%
}

\newcommand{\formulahl}[3]{\formula{\color{yellow-dark}{#1}}{#2}{#3}}

%Default options for forest

        edge label={%
            node [midway, auto, above #3, font=\scriptsize] {\color{blue-black}{#1} \hspace{0.6em} #2}%
        },
    },
}
